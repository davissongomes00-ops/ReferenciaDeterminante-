from decimal import Decimal, localcontext from typing import Union, Tuple def calculate_full_budget(area: Union[int, float]) -> Tuple[Decimal, Decimal]: """ Calculates the total budget for drywall construction based on the area. Args: area (Union[int, float]): Area of the work in square meters (m²). Returns: Tuple[Decimal, Decimal]: A tuple containing the raw subtotal and the formatted total. """ if area <= 0: raise ValueError("The area must be greater than zero.") with localcontext(): # Material unit prices PRICE_PLATE = Decimal('12.50') PRICE_STUD = Decimal('3.80') PRICE_TRACK = Decimal('3.20') PRICE_SCREW_PM = Decimal('0.05') PRICE_SCREW_LN = Decimal('0.04') PRICE_COMPOUND = Decimal('1.20') PRICE_INSULATION = Decimal('4.50') PRICE_TAPE = Decimal('0.15') PRICE_TREATMENT = Decimal('0.65') # Quantity calculations # Using -(-x // y) for ceiling division qty_plates = -(-Decimal(str(area)) // Decimal('2.88')) qty_studs = -(-Decimal(str(area)) * Decimal('2.1') // Decimal('1')) qty_tracks = -(-Decimal(str(area)) * Decimal('0.9') // Decimal('1')) qty_screws_pm = -(-Decimal(str(area)) * Decimal('18') // Decimal('1')) qty_screws_ln = -(-Decimal(str(area)) * Decimal('12') // Decimal('1')) qty_compound = (Decimal(str(area)) * Decimal('0.5')).quantize(Decimal('1')) qty_insulation = -(-Decimal(str(area)) * Decimal('1.05') // Decimal('1')) qty_tape = -(-Decimal(str(area)) * Decimal('1.6') // Decimal('1')) qty_treatment = -(-Decimal(str(area)) // Decimal('1')) # Cost calculations cost_plates = qty_plates * PRICE_PLATE cost_studs = qty_studs * PRICE_STUD cost_tracks = qty_tracks * PRICE_TRACK cost_screws_pm = qty_screws_pm * PRICE_SCREW_PM cost_screws_ln = qty_screws_ln * PRICE_SCREW_LN cost_compound = qty_compound * PRICE_COMPOUND cost_insulation = qty_insulation * PRICE_INSULATION cost_tape = qty_tape * PRICE_TAPE cost_treatment = qty_treatment * PRICE_TREATMENT # Total calculation subtotal = ( cost_plates + cost_studs + cost_tracks + cost_screws_pm + cost_screws_ln + cost_compound + cost_insulation + cost_tape + cost_treatment ) formatted_total = subtotal.quantize(Decimal('0.01')) return subtotal, formatted_total
