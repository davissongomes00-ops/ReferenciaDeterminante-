from decimal import Decimal, localcontext from typing import Union def calcular_orcamento_completo(area: Union[int, float]) -> None: """Calcula o orçamento completo Args: area (Union[int, float]): Área da obra em metros quadrados (M²) """ if area <= 0: raise ValueError("A área deve ser maior que zero") with localcontext(): P_PLACA = Decimal('12.50') P_MONTANTE = Decimal('3.80') P_RAIA = Decimal('3.20') P_PARAFUSO_PM = Decimal('0.05') P_PARAFUSO_LN = Decimal('0.04') P_MASSA = Decimal('1.20') P_MANTA = Decimal('4.50') P_FITA = Decimal('0.15') P_TRATAMENTO = Decimal('0.65') qPla = -(-area // Decimal('2.88')).to_integral_value() qMon = -(-area * Decimal('2.1') // Decimal('1')).to_integral_value() qRai = -(-area * Decimal('0.9') // Decimal('1')).to_integral_value() qPM = -(-area * Decimal('18') // Decimal('1')).to_integral_value() qLN = -(-area * Decimal('12') // Decimal('1')).to_integral_value() qMas = (area * Decimal('0.5')).quantize(Decimal('1')) qMan = -(-area * Decimal('1.05') // Decimal('1')).to_integral_value() qFit = -(-area * Decimal('1.6') // Decimal('1')).to_integral_value() qTra = -(-area // Decimal('1')).to_integral_value() tPla = qPla * P_PLACA tMon = qMon * P_MONTANTE tRai = qRai * P_RAIA tPM = qPM * P_PARAFUSO_PM tLN = qLN * P_PARAFUSO_LN tMas = qMas * P_MASSA tMan = qMan * P_MANTA tFit = qFit * P_FITA tTra = qTra * P_TRATAMENTO subtotalPladur = tPla + tMon + tRai + tPM + tLN + tMas + tMan + tFit + tTra total_formatado = subtotalPladur.quantize(Decimal('0.01')) return subtotalPladur, total_formatado
